**Leave Request Code in Python:**
```python
class LeaveRequest:
    def __init__(self, name, designation, leave_type, leave_from, leave_to, number_of_days, reason_for_leave):
        self.name = name
        self.designation = designation
        self.leave_type = leave_type
        self.leave_from = leave_from
        self.leave_to = leave_to
        self.number_of_days = number_of_days
        self.reason_for_leave = reason_for_leave

    def __str__(self):
        return f"Leave Request from {self.name} for {self.number_of_days} days from {self.leave_from} to {self.leave_to} due to {self.reason_for_leave}"

leave_request = LeaveRequest("Anshul Chauhan", "Content Writer", "Sick Leave", "2024-05-22", "2024-05-25", 4, "Health Condition")
print(leave_request)
```
